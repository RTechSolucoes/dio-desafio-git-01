Diversos "GIT BASH": 
- Echo > - (CRIAR ARQUIVO/PASTA)
- git init - INICIA O GIT NA PASTA LOCAL
- git add (.)(-a)(*)(nomeArquivo) - ADICIONA ARQUIVO/PASTA
- git rm - EXCLUIR O ARQUIVO/PASTA
- git commit
- git commit -m "(mensagem)"
- git config --list - LISTA OS DADOS DE CONFIGURAÇÃO
- Q - SAIR DO CONFIG
- git config --global - CORRIGIR ALGUMA INFORMAÇÃO DIVERGENTE DOS DADOS PESSOAIS
	--unset uner.(suaEscolha)
- git status - IMPRIMI A SITUAÇÃO DO ARQUIVO/PASTA
- git remote add origin "linkDejado(https)" - ADICIONA O REPOSITÓRIO DESEJADO
- git remote -v - LISTAR REPOSITORIOS REMOTO CADASTRADO/VINCULADO
- git push origin (main)(master)(outroQualquer) - ENVIAR O REPOSITÓRIO PARA O GITHUB
- git pull origin (main)(master)(outroQualquer) - BAIXA ATUALIZAÇÃO/ALTERAÇÃO DO REPOSITÓRIO NO GITHUB
- git clone "linkDejado(https)" - CLONA O REPOSITÓRIO DESEJADO
- git clone - VERIFICAR PARA ONDE APONTA O REPOSITÓRIO
	ls -a
- mv (pasta/arquivo) ./ (pasta/arquivo)/ - MOVE ARQUIVO/PASTA
- git --verssion
